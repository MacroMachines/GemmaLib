<desc>;
draw a rectangle to the display window.;

<args>;
x\, y = x and y coordinates in pixels (float);
p = name of preset to load (symbol);
w\, h = width and height in pixels (float);
s = size in pixels (float);
r\, g\, b\, a = rgba values (int);
c = name of color to load (symbol);
d = degree of z-axis rotation (float);
f = fill/unfill (int);
l = line width in pixels when unfilled (float);
z = z-order(drawing order) (int);
v = visible/invisible (int);
